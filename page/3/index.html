2017百度之星解题报告

07 8月 2017

2017百度之星资格赛解题报告

这次的题目比我预想中要难一点，毕竟只是一个资格赛。当然，水题还是有的，AC一道就有资格进初赛还是可以的。

不多说，先上链接2017百度之星资格赛

C题

当然是水题优先啦。
从数据范围可以看出，这是一道极为暴力的dp，参考完全背包模型，最多就是需要dp一下11个防御值时候的情况，O(bam)的复杂度，大概是10^7，妥妥的过。

AC代码：

 /*
    time:234 MS 
    Memory:3208 K
 */ 
#include<iostream>
#include<cstdio>
#include<bits/stdc++.h>
#define lowbit(i) (i&(-i))
#define LL long long
using namespace std;

int n,m;
LL INF;
vector<pair<int,int> >v;
LL dp[12][1005];
vector<pair<int,int > >sk;
void DP(){
    memset(dp,127,sizeof(dp));
    INF = dp[0][0];
    for(int i = 0; i<= 10; ++i){
        dp[i][0] = 0;
        for(int j = 0; j < m; ++j){
            int termx = sk[j].first;
            int termy = sk[j].second - i;
            if(termy <= 0)continue;
            for(int k = 0; k <= 1000; ++k){
                LL term = k >= termy ? dp[i][k - termy] : 0;
                dp[i][k] = min(dp[i][k], term + termx);    
            }
        }
    }
}
int main(){
    while(scanf("%d %d",&n,&m)!= EOF){
        int x,y;
        v.clear();sk.clear();
        for(int i = 0; i < n; ++i){
            scanf("%d %d",&x,&y);
            v.push_back(make_pair(x,y));
        } 
        for(int i = 0; i < m; ++i){
            scanf("%d %d",&x,&y);
            sk.push_back(make_pair(x,y));
        } 
        DP();
        LL ans = 0;
        bool ok = true;
        for(int i = 0; i < v.size(); ++i){
            if(dp[v[i].second][v[i].first] != INF)
                ans += dp[v[i].second][v[i].first];
            else {
                ok = false;
                break;
            } 
        }
        if(ok)printf("%lld\n",ans);
        else printf("-1\n");
    }

    return 0;
}

D题

也是一道DP水题，参考01背包模型，只是多了个输出路径，其实也是水水的，本来以为至少会卡一下空间啥的，但是暴力vector直接过了，惊讶。

AC代码：

/*
    time:   0 MS  
    memory: 1460 K
*/
#include<iostream>
#include<cstdio>
#include<bits/stdc++.h>

#define lowbit(i) (i&(-i))
#define LL long long
using namespace std;

int B,N;
int INF;
vector<pair<int,int> >v;
vector<int>ans;
int dp[1005];
struct point{
    vector<int>vec;
}p[1005];
void DP(){
    memset(dp,0,sizeof(dp));
    INF = dp[0];
    dp[0] = 0;
    for(int i = 0; i <= 1000; ++i)
        p[i].vec.clear();
    for(int i = 0;i < N; ++i){
        int x = v[i].first;
        int y = v[i].second;
        for(int j = B; j >= y; --j){
            if(dp[j] < dp[j - y] + x){
                dp[j] = dp[j - y] + x;
                p[j].vec = p[j-y].vec;
                p[j].vec.push_back(i);

            }
        }
    }
}
int main(){
    int T;
    cin>>T;
    int xu = 0;
    while(++xu <= T){
        cin>>B>>N;
        v.clear();
        int x,y;
        for(int i = 1; i <= N; ++i){
            scanf("%d %d",&x,&y);
            v.push_back(make_pair(x,y));
        }
        DP();
        int now = B;
        LL sum = 0;
        ans =p[B].vec; 

        sort(ans.begin(),ans.end());
        for(int i = 0; i < ans.size(); ++i)sum += v[ans[i]].second;    
        printf("Case #%d:\n%d %d\n",xu,dp[B],sum);
        for(int i = 0; i < ans.size(); ++i){
            if(i == ans.size() -1)printf("%d\n",ans[i]+1);
            else printf("%d ",ans[i]+1);
        }
    }
    return 0;
}

补题报告

A题

**有误，得空发现这题出错了，所有AC代码都翻车。之后再补一发。**
题目给个链接吧A题

题意大概就是要你解决            
    1）两个字符串的包含问题
    2）字符串的最大长度

  其实真正理解了题意将两个小问题拆开真的是两个不难的问题。
  第1）问其实就是KMP匹配，只不过要注意的就是字符a的特殊性，其实也就是匹配的时候字符a可以直接匹配，稍微改改KMP模板就可以实现。
  第2）问其实更水啦，就是一个简单的dp，即使暴力去搞其实也是可以的，当然啦，一不小心还是会跪的。
  菜菜的我当时就是倒在第2）问上，但一直在抠KMP，也是醉醉的。其实我当时一直在怀疑我对KMP的理解是不是出现了什么差错。

AC代码：

#include<iostream>
#include<cstdio>
#include<bits/stdc++.h>    
#define lowbit(i) (i&(-i))
#define LL long long
using namespace std;
const int maxn = 2e5+20;
int fail[maxn];
int n;
void makefail(string s){
    for(int i = 1, j= 0; s[i]; ++i){
        while(j & s[i] != s[j])j = fail[j-1];
        if(s[i] == s[j])fail[i] = ++j;
        else fail[i] = 0;
    }
}
int search(string str,string s){
    str += str;
    int ans = 0;
    for(int i = 0, j = 0;str[i]; ++i){
        while(j && str[i]!='a' && str[i] != s[j])j = fail[j-1];
        if((str[i]=='a'||str[i] == s[j]) && !s[++j]){
            ++ans;
            j = 0;
        }
    }
    return ans;
}
int num(string s) {
    s = s + s;
    int ret = 0, tmp = 0, last = -1;
    char cur = 'a';
    for(int i =0; i < n*2-1; ++i){ 
        if (cur == s[i] || s[i] == 'a' || cur == 'a') {
            tmp++;
            if (s[i] != 'a') {
                cur = s[i];
                last = i;
            }
            ret = max(tmp, ret);
            if (ret >= n) return n;
        } else {
            tmp = i-last;
            cur = s[i];
            last = i;
        }
    }
    return ret;
}
int main(){
//    freopen("a.in","r",stdin);

    string a,b;
    while(cin>>n){
        int sign = 0;
        cin>>a>>b;
        makefail(a);
        if(search(b,a) > 0)sign += 1;
        makefail(b);
        if(search(a,b) > 0)sign += 2;
        if(sign == 1)printf("Y-WIN\n");
        else if(sign == 2)printf("X-WIN\n");
        else if(sign == 3)printf("DRAW\n");
        else printf("PEACE\n");

        printf("%d\n",num(a));  
        printf("%d\n",num(b));


    }
    return 0;
}

H题

照样上链接H题

题意大致如下：
    给你n个工作时间段，每一个时间段有着开始时间，结束时间和报酬，要求你找出两个时间段满足其报酬之和为m且工作时间最短。

  很明显，这是一个dp，当然贪心也可以处理。
  我的做法有点非常规，和标程有点区别。毕竟这也是一道简单题，随便搞搞就过了。
  因为是找两个时间段且其和确定，那我们是否可以考虑折半一下，即枚举前m/2个时间段去后m/2中找符合的那个并维护一下最小时间。时间复杂度O （ nlogn + m/2 * logm/2）。显然理论上肯定是可以过的。

  因为当时处理的时候写跪了，在维护当前m-i的最小值时用了while语句，大概是被卡了，时间爆表，空间爆表，还是靠队友才过了，白白浪费了那么长的时间。之后补题时有点不甘心，优化了一下，在每个时间段排序时我直接把时间长度也作为优先级加进去，竟然就过了。hhhhh，当时怎么就没。。。

AC代码：

#include<iostream>
#include<cstdio>
#include<bits/stdc++.h>

#define lowbit(i) (i&(-i))
#define LL long long
using namespace std;
const int INF = 1e8+10;
const int maxn = 2e5+10;
struct work{
    int st;
    int et;
    int len; 
    int w;
    bool operator < (const work &a)const{
        if(w == a.w)return len < a.len;
        return w < a.w;
    }

}job[maxn];

int n,m;
int minn;
int vec[maxn];
inline void cal(work i,work j){
    int time = -1; 
    if(i.st < j.st &&i.et < j.st){
        time = i.et - i.st + j.et - j.st+2;
    }else if(i.st>j.st && i.st > j.et){
        time = i.et - i.st + j.et - j.st+2;    
    }
    if(time != -1 && time < minn)minn= time;
}

int main(){
    while(scanf("%d %d",&n,&m)!=EOF){
        minn = INF;
        for(int i = 0; i < n; ++i){
            scanf("%d %d %d",&job[i].st,&job[i].et,&job[i].w);
            job[i].len = job[i].et - job[i].st;
        }
        sort(job,job+n);

        for(int i = 0; i <n; ++i){
            vec[i] = job[i].w;
        }
        int mid = upper_bound(vec,vec+n,m/2) - vec;
        int k = upper_bound(vec,vec+n,m) - vec;

        for(int i = 0; i < mid; ++i){
            int term = lower_bound(vec + mid,vec + k, m - vec[i]) - vec;
//            while(job[term].w <= m - job[i].w){
//                if(job[term].w == m - job[i].w){
//                    cal(job[term],job[i]);    
//                }
//                ++term;
//            }
            if(job[term].w == m - job[i].w)
                cal(job[term],job[i]);
        }
        if(minn != INF)printf("%d\n",minn);
        else printf("oh no!\n");
    }

    return 0;
}

##A题##
  是个简单的贪心题，如果题目要求改成要求是不同的素数的和要就稍微需要考虑一点东西了，很遗憾，一开始就题目读错了，水题反而wa了。

B题

这是一个纯粹新手题，不管是暴力枚举还是排序后二分查找思路都是很清晰的。

C题

思路很好想，稍微写下几次操作之后就会发现6次一个循环，所以剩下的思路就是把opt次数模6然后剩下无论是暴力还是打表都是很轻松的。

D 题

  Day1的题毕竟是为了营造友好氛围的，没什么难度。写题解不过是为了完整性。
  这道题其实就是权重平衡之后最少的那个。很粗暴的计算就好。

ans = min(a,b/2,c/4)*7。

E 题

其实还是很有难度的，但对于时间不是很限制，那就是一个抖机灵的题。因为是按行按列翻转，所以很容易想到每行每列最多翻转一次。所以从行上考虑，只要保证不符合最终状态的都翻转1次，最后统计所有列是否是符合的就好。

f(1) = f(2) = f(3) = t;

[f(x+1),f(x),f(x-1)]T = [a,b,c][f(x),f(x-1),f(x-2)]T

写于大二小学期课程最后一天

14 7月 2017

  小学期课程今天下午之后就全部结束了，大部分小伙伴这周末也该回家了。也正式预示着我大二这一年的完结。
毕竟也一年了，写点东西总结一下吧。

  讲道理，进大学两年了，大学生活的点滴和我进入大学之前想象的完全不一样。没有日常的健身和运动，没有成为那个健壮的少年，没有高绩点，没有什么出彩的履历。拿得出手的还是我自己都不大看得上的几个奖状。当然我不后悔之前的时光，毕竟流逝的已经成为了过去，未来才是我生活的主旋律。

  大学以来我遇到了种种称心不称心的事：大一竞选班长时的哑口无言让我知道原来我不是想象中的无所不能，自此心中一直有了阴影，至今也没有完全好转；之后的考试，让我知道原来我的智商并不高，我可能需要付出应该付出的努力才能收获想要的结果，虽然绩点还是那么渣；一次次遇到心结无法跨过去，让我明白自己不是超人也不是主角，想要蜕变还需要更多的磨炼。
