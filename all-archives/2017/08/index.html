Snow
签到题，贪心，总是取当前数目最大的三种，各种解法。

AC代码

#include<iostream>
#include<cstdio>
#include<bits/stdc++.h>
#define mp make_pair
#define pb push_back
#define lowbit(i) (i&(-i))
#define LL long long
using namespace std;
struct point{
    int num,cnt;
    bool operator < (const point & a)const {
        if(cnt == a.cnt)return num < a.num; 
        return cnt < a.cnt;
    }
}; 
priority_queue<point>q;
vector<int >v;
int main(){
    std::ios::sync_with_stdio(false);
    int T;
    cin>>T;
    while(T--){
        int n,term;
        v.clear(); 
        while(q.size() >0)q.pop();
        cin>>n;
        for(int i = 1; i <= n ; ++i){
            cin>>term;
            v.pb(term);
        }
        sort(v.begin(), v.end());
        int temp = -1; point nn;
        for(auto i : v){
            if(temp == -1){
                temp = i;
                nn.num = i; nn.cnt = 1;
            }else if(temp != i){
                q.push(nn);
                nn.num = i; temp = i;
                nn.cnt = 1;
            }else ++nn.cnt; 
        }
        q.push(nn);
        LL ans = 0;
        while(q.size() >= 3){
            point a = q.top(); q.pop();
            point b = q.top(); q.pop();
            point c = q.top(); q.pop();
        //    cout<<a.num<<" "<<b.num <<" "<<c.num<<endl;
//每次只能取一个，不然不满足贪心正确性
//            a.cnt -= c.cnt;
//            b.cnt -= c.cnt;
//            ans += c.cnt;
            --a.cnt; --b.cnt;--c.cnt;
            ++ans;
            if(a.cnt > 0)q.push(a);
            if(b.cnt > 0)q.push(b);
            if(c.cnt > 0)q.push(c); 
        }
        cout<<ans<<endl;

    }

    return 0;
}

Plates

Plates

数学题，简单思维。
只要把n个圆的圆心围成正多边形,然后求出这个正多边形的外切圆半径再加上小圆的半径就解决了。记得特判一下n<=2的情况。

### AC代码 ###
#include<iostream>
#include<cstdio>
#include<bits/stdc++.h>
#define mp make_pair
#define pb push_back
#define lowbit(i) (i&(-i))
#define LL long long
using namespace std;
const double pi = acos(-1.0);

int main(){
    std::ios::sync_with_stdio(false);
    int n,r,R;
    while(cin>>n>>R>>r){
        if(n <= 2)cout<<(R >= r * n ? "yes":"no") <<endl;
        else{
            double du = (n-2)*1.0/n * pi;
            double m = r + r / cos(du/2);
            //cout<<m<<endl;
            if(m > R)cout<<"no"<<endl;
            else cout<<"yes"<<endl;    
        }

    } 
    return 0;
}

博弈论-k倍取数

8月 24, 2017

斐波那契取数

   训练时遇到一道博弈题，经典博弈模型-k倍取数。直接入手比较难理解，于是去找了下 “k = 2” 的斐波那契取数模型先了解一波。

原理引入

(1):齐肯多夫定理： 任意数可分解为斐波那契数列中数的和
(2)当石头总数n为斐波那契数的时候，后手必赢。

证明2参考

好像又说绕回来了。。。有点迷

k倍动态减法游戏

  原理详述 这篇博客讲的非常清晰。推荐。
将k倍动态减数问题和斐波那契数列结合起来，仿照斐波那契数列来构造a数组。

    即 a[ i + 1 ] = a[ i ] + sgma( j = 1,a[j] k < a[i]>) ( a[ j ] + 1);
            其中a[ j ] k < a[ i ]。
  引入b数组表示a[ i ] + a[ j ], 表示a数组前i项所能构造的最大数.

  简化为

     a[ i + 1] = a[ i ] + b[ j ];
    其中代表所有满足 a[j] * k < a[i] 的a[ j ]和

  不过之后的确搞通了，感谢前行者们无私的分享。

代码

const int maxn = 3e7+10;

int main(){
    int T,n,k,num = 0;
    cin >> T;
    while(++num <= T){
        cin>>n>>k;
        a[0] = b[0] = 1;
//        有误

//            if(j < 0)b[i] = a[i];
//            else b[i] = a[i] + b[j];
//        } 
        int i = 0, j = 0;  
        while(a[i]) < n{         // 构建数列  
            ++i;
            a[i] = b[i - 1] + 1;  
            while(a[j + 1] * k < a[i])++j;  
            if(k * a[j] < a[i])

            else b[i] = a[i];  
        }

            int ans;

                if(n >= a[i])n -= a[i],ans = a[i];

            }

        }
    } 
    return 0;
}

2017百度之星解题报告

8月 07, 2017

2017百度之星资格赛解题报告

这次的题目比我预想中要难一点，毕竟只是一个资格赛。当然，水题还是有的，AC一道就有资格进初赛还是可以的。

不多说，先上链接2017百度之星资格赛

C题

当然是水题优先啦。
从数据范围可以看出，这是一道极为暴力的dp，参考完全背包模型，最多就是需要dp一下11个防御值时候的情况，O(bam)的复杂度，大概是10^7，妥妥的过。

AC代码：

 /*
    time:234 MS 
    Memory:3208 K
 */ 
#include<iostream>
#include<cstdio>
#include<bits/stdc++.h>
#define lowbit(i) (i&(-i))
#define LL long long
using namespace std;

int n,m;
LL INF;
vector<pair<int,int> >v;
LL dp[12][1005];
vector<pair<int,int > >sk;
void DP(){
    memset(dp,127,sizeof(dp));
    INF = dp[0][0];
    for(int i = 0; i<= 10; ++i){
        dp[i][0] = 0;
        for(int j = 0; j < m; ++j){
            int termx = sk[j].first;
            int termy = sk[j].second - i;
            if(termy <= 0)continue;
            for(int k = 0; k <= 1000; ++k){
                LL term = k >= termy ? dp[i][k - termy] : 0;
                dp[i][k] = min(dp[i][k], term + termx);    
            }
        }
    }
}
int main(){
    while(scanf("%d %d",&n,&m)!= EOF){
        int x,y;
        v.clear();sk.clear();
        for(int i = 0; i < n; ++i){
            scanf("%d %d",&x,&y);
            v.push_back(make_pair(x,y));
        } 
        for(int i = 0; i < m; ++i){
            scanf("%d %d",&x,&y);
            sk.push_back(make_pair(x,y));
        } 
        DP();
        LL ans = 0;
        bool ok = true;
        for(int i = 0; i < v.size(); ++i){
            if(dp[v[i].second][v[i].first] != INF)
                ans += dp[v[i].second][v[i].first];
            else {
                ok = false;
                break;
            } 
        }
        if(ok)printf("%lld\n",ans);
        else printf("-1\n");
    }

    return 0;
}

D题

也是一道DP水题，参考01背包模型，只是多了个输出路径，其实也是水水的，本来以为至少会卡一下空间啥的，但是暴力vector直接过了，惊讶。

AC代码：

/*
    time:   0 MS  
    memory: 1460 K
*/
#include<iostream>
#include<cstdio>
#include<bits/stdc++.h>

#define lowbit(i) (i&(-i))
#define LL long long
using namespace std;

int B,N;
int INF;
vector<pair<int,int> >v;
vector<int>ans;
int dp[1005];
struct point{
    vector<int>vec;
}p[1005];
void DP(){
    memset(dp,0,sizeof(dp));
    INF = dp[0];
    dp[0] = 0;
    for(int i = 0; i <= 1000; ++i)
        p[i].vec.clear();
    for(int i = 0;i < N; ++i){
        int x = v[i].first;
        int y = v[i].second;
        for(int j = B; j >= y; --j){
            if(dp[j] < dp[j - y] + x){
                dp[j] = dp[j - y] + x;
                p[j].vec = p[j-y].vec;
                p[j].vec.push_back(i);

            }
        }
    }
}
int main(){
    int T;
    cin>>T;
    int xu = 0;
    while(++xu <= T){
        cin>>B>>N;
        v.clear();
        int x,y;
        for(int i = 1; i <= N; ++i){
            scanf("%d %d",&x,&y);
            v.push_back(make_pair(x,y));
        }
        DP();
        int now = B;
        LL sum = 0;
        ans =p[B].vec; 

        sort(ans.begin(),ans.end());
        for(int i = 0; i < ans.size(); ++i)sum += v[ans[i]].second;    
        printf("Case #%d:\n%d %d\n",xu,dp[B],sum);
        for(int i = 0; i < ans.size(); ++i){
            if(i == ans.size() -1)printf("%d\n",ans[i]+1);
            else printf("%d ",ans[i]+1);
        }
    }
    return 0;
}
