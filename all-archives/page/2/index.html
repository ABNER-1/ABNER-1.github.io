            point a = q.top(); q.pop();

只要把n个圆的圆心围成正多边形,然后求出这个正多边形的外切圆半径再加上小圆的半径就解决了。记得特判一下n<=2的情况。

        if(n <= 2)cout<<(R >= r * n ? "yes":"no") <<endl;
        else{
            double du = (n-2)*1.0/n * pi;
            double m = r + r / cos(du/2);

        }

博弈论-k倍取数

8月 24, 2017

斐波那契取数

   训练时遇到一道博弈题，经典博弈模型-k倍取数。直接入手比较难理解，于是去找了下 “k = 2” 的斐波那契取数模型先了解一波。

原理引入

(1):齐肯多夫定理： 任意数可分解为斐波那契数列中数的和
(2)当石头总数n为斐波那契数的时候，后手必赢。

证明2参考

好像又说绕回来了。。。有点迷

k倍动态减法游戏

  原理详述 这篇博客讲的非常清晰。推荐。
将k倍动态减数问题和斐波那契数列结合起来，仿照斐波那契数列来构造a数组。

    即 a[ i + 1 ] = a[ i ] + sgma( j = 1,a[j] k < a[i]>) ( a[ j ] + 1);
            其中a[ j ] k < a[ i ]。
  引入b数组表示a[ i ] + a[ j ], 表示a数组前i项所能构造的最大数.

  简化为

     a[ i + 1] = a[ i ] + b[ j ];
    其中代表所有满足 a[j] * k < a[i] 的a[ j ]和

  不过之后的确搞通了，感谢前行者们无私的分享。

C题

当然是水题优先啦。
从数据范围可以看出，这是一道极为暴力的dp，参考完全背包模型，最多就是需要dp一下11个防御值时候的情况，O(bam)的复杂度，大概是10^7，妥妥的过。

AC代码：

 /*
    time:234 MS 
    Memory:3208 K
 */ 
#include<iostream>
#include<cstdio>
#include<bits/stdc++.h>
#define lowbit(i) (i&(-i))
#define LL long long
using namespace std;

int n,m;
LL INF;
vector<pair<int,int> >v;
LL dp[12][1005];
vector<pair<int,int > >sk;
void DP(){
    memset(dp,127,sizeof(dp));
    INF = dp[0][0];
    for(int i = 0; i<= 10; ++i){
        dp[i][0] = 0;
        for(int j = 0; j < m; ++j){
            int termx = sk[j].first;
            int termy = sk[j].second - i;
            if(termy <= 0)continue;
            for(int k = 0; k <= 1000; ++k){
                LL term = k >= termy ? dp[i][k - termy] : 0;
                dp[i][k] = min(dp[i][k], term + termx);    
            }
        }
    }
}
int main(){
    while(scanf("%d %d",&n,&m)!= EOF){
        int x,y;
        v.clear();sk.clear();
        for(int i = 0; i < n; ++i){
            scanf("%d %d",&x,&y);
            v.push_back(make_pair(x,y));
        } 
        for(int i = 0; i < m; ++i){
            scanf("%d %d",&x,&y);
            sk.push_back(make_pair(x,y));
        } 
        DP();
        LL ans = 0;
        bool ok = true;
        for(int i = 0; i < v.size(); ++i){
            if(dp[v[i].second][v[i].first] != INF)
                ans += dp[v[i].second][v[i].first];
            else {
                ok = false;
                break;
            } 
        }
        if(ok)printf("%lld\n",ans);
        else printf("-1\n");
    }

    return 0;
}

H题

照样上链接H题

题意大致如下：
    给你n个工作时间段，每一个时间段有着开始时间，结束时间和报酬，要求你找出两个时间段满足其报酬之和为m且工作时间最短。

  很明显，这是一个dp，当然贪心也可以处理。
  我的做法有点非常规，和标程有点区别。毕竟这也是一道简单题，随便搞搞就过了。
  因为是找两个时间段且其和确定，那我们是否可以考虑折半一下，即枚举前m/2个时间段去后m/2中找符合的那个并维护一下最小时间。时间复杂度O （ nlogn + m/2 * logm/2）。显然理论上肯定是可以过的。

  因为当时处理的时候写跪了，在维护当前m-i的最小值时用了while语句，大概是被卡了，时间爆表，空间爆表，还是靠队友才过了，白白浪费了那么长的时间。之后补题时有点不甘心，优化了一下，在每个时间段排序时我直接把时间长度也作为优先级加进去，竟然就过了。hhhhh，当时怎么就没。。。

AC代码：

#include<iostream>
#include<cstdio>
#include<bits/stdc++.h>

#define lowbit(i) (i&(-i))
#define LL long long
using namespace std;
const int INF = 1e8+10;
const int maxn = 2e5+10;
struct work{
    int st;
    int et;
    int len; 
    int w;
    bool operator < (const work &a)const{
        if(w == a.w)return len < a.len;
        return w < a.w;
    }

}job[maxn];

int n,m;
int minn;
int vec[maxn];
inline void cal(work i,work j){
    int time = -1; 
    if(i.st < j.st &&i.et < j.st){
        time = i.et - i.st + j.et - j.st+2;
    }else if(i.st>j.st && i.st > j.et){
        time = i.et - i.st + j.et - j.st+2;    
    }
    if(time != -1 && time < minn)minn= time;
}

int main(){
    while(scanf("%d %d",&n,&m)!=EOF){
        minn = INF;
        for(int i = 0; i < n; ++i){
            scanf("%d %d %d",&job[i].st,&job[i].et,&job[i].w);
            job[i].len = job[i].et - job[i].st;
        }
        sort(job,job+n);

        for(int i = 0; i <n; ++i){
            vec[i] = job[i].w;
        }

最近真的是糟心的事一堆堆

  先是因为感冒的缘故难受了好几天，这几天稍微好过了一点，却因为一点小插曲度过了一个很奇妙的周五。
