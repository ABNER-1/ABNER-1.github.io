签到题，贪心，总是取当前数目最大的三种，各种解法。

AC代码

#include<iostream>
#include<cstdio>
#include<bits/stdc++.h>
#define mp make_pair
#define pb push_back
#define lowbit(i) (i&(-i))
#define LL long long
using namespace std;
struct point{
    int num,cnt;
    bool operator < (const point & a)const {
        if(cnt == a.cnt)return num < a.num; 
        return cnt < a.cnt;
    }
}; 
priority_queue<point>q;
vector<int >v;
int main(){
    std::ios::sync_with_stdio(false);
    int T;
    cin>>T;
    while(T--){
        int n,term;
        v.clear(); 
        while(q.size() >0)q.pop();
        cin>>n;
        for(int i = 1; i <= n ; ++i){
            cin>>term;
            v.pb(term);
        }
        sort(v.begin(), v.end());
        int temp = -1; point nn;
        for(auto i : v){
            if(temp == -1){
                temp = i;
                nn.num = i; nn.cnt = 1;
            }else if(temp != i){
                q.push(nn);
                nn.num = i; temp = i;
                nn.cnt = 1;
            }else ++nn.cnt; 
        }
        q.push(nn);
        LL ans = 0;
        while(q.size() >= 3){
            point a = q.top(); q.pop();
            point b = q.top(); q.pop();
            point c = q.top(); q.pop();
        //    cout<<a.num<<" "<<b.num <<" "<<c.num<<endl;
//每次只能取一个，不然不满足贪心正确性
//            a.cnt -= c.cnt;
//            b.cnt -= c.cnt;
//            ans += c.cnt;
            --a.cnt; --b.cnt;--c.cnt;
            ++ans;
            if(a.cnt > 0)q.push(a);
            if(b.cnt > 0)q.push(b);
            if(c.cnt > 0)q.push(c); 
        }
        cout<<ans<<endl;

    }

    return 0;
}

数学题，简单思维。
只要把n个圆的圆心围成正多边形,然后求出这个正多边形的外切圆半径再加上小圆的半径就解决了。记得特判一下n<=2的情况。

### AC代码 ###
#include<iostream>
#include<cstdio>
#include<bits/stdc++.h>
#define mp make_pair
#define pb push_back
#define lowbit(i) (i&(-i))
#define LL long long
using namespace std;
const double pi = acos(-1.0);

int main(){
    std::ios::sync_with_stdio(false);
    int n,r,R;
    while(cin>>n>>R>>r){
        if(n <= 2)cout<<(R >= r * n ? "yes":"no") <<endl;
        else{
            double du = (n-2)*1.0/n * pi;
            double m = r + r / cos(du/2);
            //cout<<m<<endl;
            if(m > R)cout<<"no"<<endl;
            else cout<<"yes"<<endl;    
        }

    } 
    return 0;
}
