2017百度之星解题报告

07 8月 2017

2017百度之星资格赛解题报告

这次的题目比我预想中要难一点，毕竟只是一个资格赛。当然，水题还是有的，AC一道就有资格进初赛还是可以的。

不多说，先上链接2017百度之星资格赛

C题

当然是水题优先啦。
从数据范围可以看出，这是一道极为暴力的dp，参考完全背包模型，最多就是需要dp一下11个防御值时候的情况，O(bam)的复杂度，大概是10^7，妥妥的过。

AC代码：

 /*
    time:234 MS 
    Memory:3208 K
 */ 
#include<iostream>
#include<cstdio>
#include<bits/stdc++.h>
#define lowbit(i) (i&(-i))
#define LL long long
using namespace std;

int n,m;
LL INF;
vector<pair<int,int> >v;
LL dp[12][1005];
vector<pair<int,int > >sk;
void DP(){
    memset(dp,127,sizeof(dp));
    INF = dp[0][0];
    for(int i = 0; i<= 10; ++i){
        dp[i][0] = 0;
        for(int j = 0; j < m; ++j){
            int termx = sk[j].first;
            int termy = sk[j].second - i;
            if(termy <= 0)continue;
            for(int k = 0; k <= 1000; ++k){
                LL term = k >= termy ? dp[i][k - termy] : 0;
                dp[i][k] = min(dp[i][k], term + termx);    
            }
        }
    }
}
int main(){
    while(scanf("%d %d",&n,&m)!= EOF){
        int x,y;
        v.clear();sk.clear();
        for(int i = 0; i < n; ++i){
            scanf("%d %d",&x,&y);
            v.push_back(make_pair(x,y));
        } 
        for(int i = 0; i < m; ++i){
            scanf("%d %d",&x,&y);
            sk.push_back(make_pair(x,y));
        } 
        DP();
        LL ans = 0;
        bool ok = true;
        for(int i = 0; i < v.size(); ++i){
            if(dp[v[i].second][v[i].first] != INF)
                ans += dp[v[i].second][v[i].first];
            else {
                ok = false;
                break;
            } 
        }
        if(ok)printf("%lld\n",ans);
        else printf("-1\n");
    }

    return 0;
}

D题

也是一道DP水题，参考01背包模型，只是多了个输出路径，其实也是水水的，本来以为至少会卡一下空间啥的，但是暴力vector直接过了，惊讶。

AC代码：

/*
    time:   0 MS  
    memory: 1460 K
*/
#include<iostream>
#include<cstdio>
#include<bits/stdc++.h>

#define lowbit(i) (i&(-i))
#define LL long long
using namespace std;

int B,N;
int INF;
vector<pair<int,int> >v;
vector<int>ans;
int dp[1005];
struct point{
    vector<int>vec;
}p[1005];
void DP(){
    memset(dp,0,sizeof(dp));
    INF = dp[0];
    dp[0] = 0;
    for(int i = 0; i <= 1000; ++i)
        p[i].vec.clear();
    for(int i = 0;i < N; ++i){
        int x = v[i].first;
        int y = v[i].second;
        for(int j = B; j >= y; --j){
            if(dp[j] < dp[j - y] + x){
                dp[j] = dp[j - y] + x;
                p[j].vec = p[j-y].vec;
                p[j].vec.push_back(i);

            }
        }
    }
}
int main(){
    int T;
    cin>>T;
    int xu = 0;
    while(++xu <= T){
        cin>>B>>N;
        v.clear();
        int x,y;
        for(int i = 1; i <= N; ++i){
            scanf("%d %d",&x,&y);
            v.push_back(make_pair(x,y));
        }
        DP();
        int now = B;
        LL sum = 0;
        ans =p[B].vec; 

        sort(ans.begin(),ans.end());
        for(int i = 0; i < ans.size(); ++i)sum += v[ans[i]].second;    
        printf("Case #%d:\n%d %d\n",xu,dp[B],sum);
        for(int i = 0; i < ans.size(); ++i){
            if(i == ans.size() -1)printf("%d\n",ans[i]+1);
            else printf("%d ",ans[i]+1);
        }
    }
    return 0;
}

B题

这是一道出题人得气死的题。
本来看了数据范围和时间限制，根本就是个不可做题，全局最小割加优化大概才可以过。但在乱搞面前，什么都是浮云。
真心，出题人数据一开始卡死了认认真真写SW最小割的，但败在各种瞎搞的过题姿势下。
你能想像那种求每个点的边权和最小的就是正解？？？这么简单的漏洞，出题人都没卡，绝壁是道假题，吓得我当时题都不敢看。。
各种代码就不贴了，正解我还没搞懂，有空补，不过估计是没空了。

E题

这是一道我认为的最有价值的题。学到了很多，难度应该是真B题之外最难的。
不过模型很常见：卡特兰数-百度百科。
当时我是一点头绪都没有的，结束之后参考了几位大牛的博客，get了不少。
参考大牛OZY
另一个大牛的
两位大牛的思路类似，又稍有区别，实现大同小异，值得学习一波。
AC代码就算了，这题也就思路很不错，仿照两位大牛的博客试试就行。

A题

这题只是匆匆一瞥，大致的思路类似于凸包吧，看dalao们的AC代码都是求最小环，最后设边权为1的去跑floyd，也没深究，不是很感兴趣，有空会补上。
缺点太多，要补的东西成堆，都麻木了。
