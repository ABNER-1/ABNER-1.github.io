题目给个链接吧A题

题意大概就是要你解决            
    1）两个字符串的包含问题
    2）字符串的最大长度

  其实真正理解了题意将两个小问题拆开真的是两个不难的问题。
  第1）问其实就是KMP匹配，只不过要注意的就是字符a的特殊性，其实也就是匹配的时候字符a可以直接匹配，稍微改改KMP模板就可以实现。
  第2）问其实更水啦，就是一个简单的dp，即使暴力去搞其实也是可以的，当然啦，一不小心还是会跪的。
  菜菜的我当时就是倒在第2）问上，但一直在抠KMP，也是醉醉的。其实我当时一直在怀疑我对KMP的理解是不是出现了什么差错。

AC代码：

#include<iostream>
#include<cstdio>
#include<bits/stdc++.h>    
#define lowbit(i) (i&(-i))
#define LL long long
using namespace std;
const int maxn = 2e5+20;
int fail[maxn];
int n;
void makefail(string s){
    for(int i = 1, j= 0; s[i]; ++i){
        while(j & s[i] != s[j])j = fail[j-1];
        if(s[i] == s[j])fail[i] = ++j;
        else fail[i] = 0;
    }
}
int search(string str,string s){
    str += str;
    int ans = 0;
    for(int i = 0, j = 0;str[i]; ++i){
        while(j && str[i]!='a' && str[i] != s[j])j = fail[j-1];
        if((str[i]=='a'||str[i] == s[j]) && !s[++j]){
            ++ans;
            j = 0;
        }
    }
    return ans;
}
int num(string s) {
    s = s + s;
    int ret = 0, tmp = 0, last = -1;
    char cur = 'a';
    for(int i =0; i < n*2-1; ++i){ 
        if (cur == s[i] || s[i] == 'a' || cur == 'a') {
            tmp++;
            if (s[i] != 'a') {
                cur = s[i];
                last = i;
            }
            ret = max(tmp, ret);
            if (ret >= n) return n;
        } else {
            tmp = i-last;
            cur = s[i];
            last = i;
        }
    }
    return ret;
}
int main(){
//    freopen("a.in","r",stdin);

    string a,b;
    while(cin>>n){
        int sign = 0;
        cin>>a>>b;
        makefail(a);
        if(search(b,a) > 0)sign += 1;
        makefail(b);
        if(search(a,b) > 0)sign += 2;
        if(sign == 1)printf("Y-WIN\n");
        else if(sign == 2)printf("X-WIN\n");
        else if(sign == 3)printf("DRAW\n");
        else printf("PEACE\n");

        printf("%d\n",num(a));  
        printf("%d\n",num(b));


    }
    return 0;
}

H题

照样上链接H题

题意大致如下：
    给你n个工作时间段，每一个时间段有着开始时间，结束时间和报酬，要求你找出两个时间段满足其报酬之和为m且工作时间最短。

  很明显，这是一个dp，当然贪心也可以处理。
  我的做法有点非常规，和标程有点区别。毕竟这也是一道简单题，随便搞搞就过了。
  因为是找两个时间段且其和确定，那我们是否可以考虑折半一下，即枚举前m/2个时间段去后m/2中找符合的那个并维护一下最小时间。时间复杂度O （ nlogn + m/2 * logm/2）。显然理论上肯定是可以过的。

  因为当时处理的时候写跪了，在维护当前m-i的最小值时用了while语句，大概是被卡了，时间爆表，空间爆表，还是靠队友才过了，白白浪费了那么长的时间。之后补题时有点不甘心，优化了一下，在每个时间段排序时我直接把时间长度也作为优先级加进去，竟然就过了。hhhhh，当时怎么就没。。。

#include<iostream>
#include<cstdio>
#include<bits/stdc++.h>

#define lowbit(i) (i&(-i))
#define LL long long
using namespace std;
const int INF = 1e8+10;
const int maxn = 2e5+10;
struct work{
    int st;
    int et;
    int len; 
    int w;
    bool operator < (const work &a)const{
        if(w == a.w)return len < a.len;
        return w < a.w;
    }

}job[maxn];

int n,m;
int minn;
int vec[maxn];
inline void cal(work i,work j){
    int time = -1; 
    if(i.st < j.st &&i.et < j.st){
        time = i.et - i.st + j.et - j.st+2;
    }else if(i.st>j.st && i.st > j.et){
        time = i.et - i.st + j.et - j.st+2;    
    }
    if(time != -1 && time < minn)minn= time;
}

int main(){
    while(scanf("%d %d",&n,&m)!=EOF){
        minn = INF;
        for(int i = 0; i < n; ++i){
            scanf("%d %d %d",&job[i].st,&job[i].et,&job[i].w);
            job[i].len = job[i].et - job[i].st;
        }
        sort(job,job+n);

        for(int i = 0; i <n; ++i){
            vec[i] = job[i].w;
        }
        int mid = upper_bound(vec,vec+n,m/2) - vec;
        int k = upper_bound(vec,vec+n,m) - vec;

        for(int i = 0; i < mid; ++i){
            int term = lower_bound(vec + mid,vec + k, m - vec[i]) - vec;
//            while(job[term].w <= m - job[i].w){
//                if(job[term].w == m - job[i].w){
//                    cal(job[term],job[i]);    
//                }
//                ++term;
//            }
            if(job[term].w == m - job[i].w)
                cal(job[term],job[i]);
        }
        if(minn != INF)printf("%d\n",minn);
        else printf("oh no!\n");
    }

    return 0;
}
